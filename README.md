# lvmjit

compilation lua fucntion during execution of lua program at run time.

### tutorail
~~~.lua
local lvmjit = require "lvmjit.core"

local function func(...)
    -- do something..
end

local cfunc = lvmjit.compile(func)
cfunc(...)
~~~

### benchmark
test lua case in `test.lua`

~~~.lua
local function foo(a, b)
    local sum = a + b
    local d = 1
    for i=1,400000000 do
        sum = sum + d
    end
    return sum
end
~~~
